    "from pathlib import Path\n",
    "import sys\n",
    "\n",
    "\n",
    "def find_repo_root(start=Path.cwd()):\n",
    "    for path in (start, *start.parents):\n",
    "        if (path / \"pyproject.toml\").exists():\n",
    "            return path\n",
    "    return start\n",
    "\n",
    "\n",
    "NOTEBOOK_DIR = Path.cwd()\n",
    "REPO_ROOT = find_repo_root(NOTEBOOK_DIR)\n",
    "if str(REPO_ROOT) not in sys.path:\n",
    "    sys.path.insert(0, str(REPO_ROOT))\n",
    "\n",
    "import matplotlib.pyplot as plt\n",
    "import TOVsolver.main as main\n",
    "from TOVsolver.unit import km, Msun\n",
    "\n",
    "\n",
    "def find_test_eos():\n",
    "    candidates = [\n",
    "        NOTEBOOK_DIR / \"Test_EOS.csv\",\n",
    "        NOTEBOOK_DIR / \"Test_Case\" / \"Test_EOS.csv\",\n",
    "        REPO_ROOT / \"Test_Case\" / \"Test_EOS.csv\",\n",
    "    ]\n",
    "    checked = []\n",
    "    for candidate in candidates:\n",
    "        candidate = candidate.resolve()\n",
    "        if candidate in checked:\n",
    "            continue\n",
    "        checked.append(candidate)\n",
    "        if candidate.exists():\n",
    "            return candidate\n",
    "    paths = \", \".join(str(path) for path in checked)\n",
    "    raise FileNotFoundError(f\"Could not find Test_EOS.csv. Checked: {paths}\")\n",
    "\n",
    "\n",
    "TEST_EOS_PATH = find_test_eos()\n",
    "\n",
    "# Main has two functions\n",
    "#   OutputMR() returns mass and radius\n",
    "#   OutputMRT() returns radius, mass, and tidal deformability\n",
    "#   OutputC_s() returns the speed of sound and density grid\n",
    "# The input can be either a CSV path or density/pressure arrays.\n",
    "MR = main.OutputMR(str(TEST_EOS_PATH)).T\n",
    "\n",
    "fig, ax = plt.subplots(1, 1, figsize=(9, 6))\n",
    "ax.plot(MR[1] / km, MR[0] / Msun, lw=2)\n",
    "ax.set_ylabel(r\"M [$M_{\\odot}$]\", fontsize=16)\n",
    "ax.set_xlabel(\"R [km]\", fontsize=16)\n",
    "ax.set_xlim(8.0, 20.0)\n",
    "ax.set_ylim(0, 3)\n",
    "ax.tick_params(top=1, right=1, which=\"both\", direction=\"in\", labelsize=14)\n",
    "ax.tick_params(top=1, right=1, which=\"both\", direction=\"in\", labelsize=14)\n",
    "fig.tight_layout()\n",
    "plt.show()"
